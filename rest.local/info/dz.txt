Исправить работу пагинации
ограничение на кол-во накладывается на id картинок
SELECT 
	DISTINCT G.id
FROM 
	Gallery G 
	JOIN Literals L ON L.id_entity = G.id
	JOIN Langs A ON L.id_lang = A.id
LIMIT 
	0, 4

затем при выборке описаний накладывается условие
на отобранные id

SELECT 
	G.id,
	G.filename,
	G.moment,
	A.iso639_1,
	L.txt AS descr
FROM 
	Gallery G 
	JOIN Literals L ON L.id_entity = G.id
	JOIN Langs A ON L.id_lang = A.id
WHERE
    G.id IN (
		99574855992606720,
		99574855992606721,
		99582087073366017,
		99582087073366021)
